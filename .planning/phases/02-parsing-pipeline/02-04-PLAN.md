```typescript
export function detectAnomalies(events: LogEvent[], stepContexts: StepContext[]): Anomaly[] {
  return events
    .filter(e => e.level === 'ERROR')
    .map(e => {
      const ctx = stepContexts.find(
        s => e.fileLineNumber >= s.startLine && e.fileLineNumber <= s.endLine
      );
      return {
        id: String(e.fileLineNumber),
        type: 'ERROR' as const,
        message: e.message,
        stacktrace: e.stacktrace,
        step: ctx ? ctx.step.name : '_init_',
        phase: ctx ? ctx.phase : 'Precondition' as const,
        file: '',              // engine.ts populates with actual file path
        line: e.fileLineNumber,
        sourceClass: e.sourceClass,
        sourceMethod: e.sourceMethod,
        sourceLine: e.sourceLine,
        exceptionType: e.exceptionType,
        timestamp: e.timestamp,
      };
    });
}
```

// test/suite/aggregator.test.ts
import * as assert from 'assert';
import * as path from 'path';
import * as fs from 'fs';
import { aggregateAnomalies, rankAnomalies, hashKey } from '../../src/core/aggregator';
import { parseLog } from '../../src/core/parser';
import { parseFeature } from '../../src/core/featureParser';
import { extractSteps } from '../../src/core/stepExtractor';
import { detectAnomalies } from '../../src/core/detector';
import type { Anomaly } from '../../src/types';

function makeAnomaly(overrides: Partial<Anomaly> = {}): Anomaly {
  return {
    id: '1',
    type: 'ERROR',
    message: 'default message',
    step: 'defaultStep',
    phase: 'TestCase',
    file: '',
    line: 1,
    sourceClass: 'DefaultClass',
    sourceMethod: 'DefaultMethod',
    sourceLine: 1,
    timestamp: '2025-07-03 13:52:41,220',
    ...overrides,
  };
}

suite('Anomaly aggregation', () => {
  test('aggregateAnomalies returns empty array for empty input', () => {
    assert.deepStrictEqual(aggregateAnomalies([]), []);
  });

  test('two identical anomalies produce one AggregatedAnomaly with occurrences=2', () => {
    const a1 = makeAnomaly({ id: '1', line: 1 });
    const a2 = makeAnomaly({ id: '2', line: 2 });
    const result = aggregateAnomalies([a1, a2]);
    assert.strictEqual(result.length, 1);
    assert.strictEqual(result[0].occurrences, 2);
  });

  test('anomalies with different step produce separate AggregatedAnomaly records', () => {
    const a1 = makeAnomaly({ step: 'StepA', id: '1' });
    const a2 = makeAnomaly({ step: 'StepB', id: '2' });
    const result = aggregateAnomalies([a1, a2]);
    assert.strictEqual(result.length, 2);
  });

  test('AggregatedAnomaly.key is a 64-char SHA-256 hex string', () => {
    const result = aggregateAnomalies([makeAnomaly()]);
    assert.strictEqual(result[0].key.length, 64);
    assert.ok(/^[0-9a-f]{64}$/.test(result[0].key), 'key must be lowercase hex');
  });

  test('hashKey returns 64-char hex for any input', () => {
    const h = hashKey('test input');
    assert.strictEqual(h.length, 64);
    assert.ok(/^[0-9a-f]{64}$/.test(h));
  });

  test('rankAnomalies returns empty array for empty input', () => {
    assert.deepStrictEqual(rankAnomalies([]), []);
  });

  test('rankAnomalies sorts by firstOccurrence.timestamp ascending', () => {
    const a1 = makeAnomaly({ timestamp: '2025-07-03 13:52:42,000', id: '2', line: 2 });
    const a2 = makeAnomaly({ timestamp: '2025-07-03 13:52:41,000', id: '1', line: 1 });
    const grouped = aggregateAnomalies([a1, a2]);
    const ranked = rankAnomalies(grouped);
    assert.strictEqual(ranked[0].firstOccurrence.timestamp, '2025-07-03 13:52:41,000');
