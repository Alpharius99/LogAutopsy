Implement runBatch (primary export):
```typescript
export async function runBatch(batchFolderUri: vscode.Uri): Promise<BatchResult> {
  const results: AnalysisResult[] = [];
  const errors: { testCase: string; error: string }[] = [];

  let entries: [string, vscode.FileType][];
  try {
    entries = await vscode.workspace.fs.readDirectory(batchFolderUri);
  } catch (err) {
    return { results, errors: [{ testCase: batchFolderUri.fsPath, error: String(err) }] };
  }

  for (const [name, type] of entries) {
    if (type !== vscode.FileType.Directory) { continue; }
    if (PHASE_SUBFOLDER.test(name)) { continue; }

    const folderUri = vscode.Uri.joinPath(batchFolderUri, name);
    try {
      const result = await analyzeTestCase(folderUri, name);
      results.push(result);
    } catch (err) {
      errors.push({ testCase: name, error: String(err) });
    }
  }

  return { results, errors };
}
```

Implement analyzeTestCase (private helper):
```typescript
async function analyzeTestCase(folderUri: vscode.Uri, testCase: string): Promise<AnalysisResult> {
  const entries = await vscode.workspace.fs.readDirectory(folderUri);

  // Find log file at folder root: filename contains underscore and ends with .log
  const logEntry = entries.find(([n, t]) => t === vscode.FileType.File && /_.+\.log$/.test(n));
  const featureEntry = entries.find(([n, t]) => t === vscode.FileType.File && n.endsWith('.feature'));

  if (!logEntry) {
    throw new Error(`No log file found in test case folder: ${testCase}`);
  }
  if (!featureEntry) {
    throw new Error(`No .feature file found in test case folder: ${testCase}`);
  }

  const logUri = vscode.Uri.joinPath(folderUri, logEntry[0]);
  const featureUri = vscode.Uri.joinPath(folderUri, featureEntry[0]);

  const logBytes = await vscode.workspace.fs.readFile(logUri);
  const featureBytes = await vscode.workspace.fs.readFile(featureUri);
  const logContent = Buffer.from(logBytes).toString('utf-8');
  const featureContent = Buffer.from(featureBytes).toString('utf-8');

  const events = parseLog(logContent);
  const stepPhaseMap = parseFeature(featureContent);
  const stepContexts = extractSteps(events, stepPhaseMap);
  const anomalies = detectAnomalies(events, stepContexts);

  // Populate file path on each anomaly (engine.ts knows the URI)
  const anomaliesWithFile = anomalies.map(a => ({ ...a, file: logUri.fsPath }));

  const aggregated = aggregateAnomalies(anomaliesWithFile);
  const ranked = rankAnomalies(aggregated);

  return { testCase, anomalies: ranked };
}
```

```typescript
export async function runAnalysis(): Promise<void> {
  const uris = await vscode.window.showOpenDialog({
    canSelectFolders: true,
    canSelectFiles: false,
    canSelectMany: false,
    openLabel: 'Select BatchRun Folder',
  });
  if (!uris || uris.length === 0) { return; }

  await vscode.window.withProgress(
    { location: vscode.ProgressLocation.Notification, title: 'LogAutopsy: Analyzing...' },
    async () => {
      const batchResult = await runBatch(uris[0]);
      const total = batchResult.results.reduce((n, r) => n + r.anomalies.length, 0);
      const errCount = batchResult.errors.length;
      const msg = errCount > 0
        ? `LogAutopsy: ${batchResult.results.length} test case(s) analyzed, ${total} anomaly group(s) found. ${errCount} error(s).`
        : `LogAutopsy: ${batchResult.results.length} test case(s) analyzed, ${total} anomaly group(s) found.`;
      vscode.window.showInformationMessage(msg);
    }
  );
}
```

| T-02-05-01 | Information Disclosure | engine.ts reads any folder the analyst selects | accept | The analyst explicitly selects the folder. | T-02-05-02 | Denial of Service | readDirectory on a very large folder with many subfolders | accept | Phase constraint: batch runs contain 1–10 test cases per CLAUDE.md. | T-02-05-03 | Denial of Service | UTF-8 decoding of a non-UTF-8 log file | mitigate | Use Buffer.from(bytes).toString('utf-8') — Node.js replaces malformed sequences with the Unicode replacement character U+FFFD rather than throwing. The pipeline continues; log events with replacement characters are treated as INFO/DEBUG (not ERROR) and do not produce anomalies. |
  | T-02-05-04 | Tampering | Crafted .feature file causing Gherkin parser to hang | accept | Feature files are internal test artifacts produced by the team's own test framework. No external user input reaches the Gherkin parser. @cucumber/gherkin is a well-maintained official package. |
