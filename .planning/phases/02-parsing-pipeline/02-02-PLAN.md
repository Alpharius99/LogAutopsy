export interface LogEvent {
  timestamp: string;

  level: 'ERROR' | 'WARN' | 'INFO' | 'DEBUG';

    assert.strictEqual(events[0].sourceMethod, 'WaitForTask');
    assert.strictEqual(events[0].sourceLine, 31);
    assert.strictEqual(events[0].thread, 14);
    assert.strictEqual(events[0].fileLineNumber, 1);
  });

  test('parses STANDARD_SOURCE log line correctly', () => {
    const line = '2025-07-03 13:52:40,680 [14] DEBUG GherkinExecutor\\ExecuteStep:187 - Next test step.';
    const events = parseLog(line);
    assert.strictEqual(events.length, 1);
    assert.strictEqual(events[0].sourceClass, 'GherkinExecutor');
    assert.strictEqual(events[0].sourceMethod, 'ExecuteStep');
    assert.strictEqual(events[0].sourceLine, 187);
    assert.strictEqual(events[0].exceptionType, undefined);
  });

  test('continuation lines accumulate into raw including blank lines', () => {
    const content = [
      '2025-07-03 13:52:41,220 [14] INFO Foo\\Bar:1 - msg',
      '  stack line',
      '',
      'next continuation',
      '2025-07-03 13:52:41,221 [14] INFO Foo\\Bar:2 - msg2',
    ].join('\n');
    const events = parseLog(content);
    assert.strictEqual(events.length, 2);
    assert.ok(events[0].raw.includes('\n  stack line'), 'raw must include first continuation');
    assert.ok(events[0].raw.includes('\n\nnext continuation'), 'raw must include blank + next continuation');
  });

  test('ERROR continuation lines accumulate into stacktrace; non-ERROR do not', () => {
    const errorLine = '2025-07-03 13:52:41,220 [14] ERROR A|Ex in B:1 - msg';
    const debugLine = '2025-07-03 13:52:41,221 [14] DEBUG Foo\\Bar:2 - msg2';
    const content = errorLine + '\n  at StackFrame\n' + debugLine + '\n  at NonError';
    const events = parseLog(content);
    assert.strictEqual(events.length, 2);
    assert.ok(events[0].stacktrace?.includes('at StackFrame'), 'ERROR stacktrace must be set');
